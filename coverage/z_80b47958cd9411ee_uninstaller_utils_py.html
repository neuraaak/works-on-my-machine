38def get_files_to_remove(target_path: Path) -> list[str]: 

39 """ 

40 Get list of files and directories to remove for progress tracking. 

41 

42 Args: 

43 target_path: Target installation directory 

44 

45 Returns: 

46 List[str]: List of relative file and directory paths to remove 

47 

48 Raises: 

49 FileScanError: If file list generation fails 

50 DirectoryAccessError: If directory access fails 

51 """ 

52 try: 

53 # Input validation 

54 if not target_path: 

55 raise FileScanError( 

56 operation="list_generation", 

57 target_path="", 

58 message="Target path cannot be empty", 

59 details="Invalid target path provided for file list generation", 

60 ) 

61 

62 files_to_remove = [] 

63 

64 if not target_path.exists(): 

65 return files_to_remove 

66 

67 try: 

68 # Check if we have permission to access the directory 

69 if not target_path.is_dir(): 

70 raise DirectoryAccessError( 

71 operation="list_generation", 

72 directory_path=str(target_path), 

73 message="Target path is not a directory", 

74 details=f"Path exists but is not a directory: {target_path}", 

75 ) 

76 

77 # Get all files and directories recursively 

78 for item_path in target_path.rglob("*"): 

79 try: 

80 if item_path.is_file(): 

81 # Add file with relative path 

82 relative_path = item_path.relative_to(target_path) 

83 files_to_remove.append(str(relative_path)) 

84 elif item_path.is_dir(): 

85 # Add directory with relative path (keep trailing slash for directories) 

86 relative_path = item_path.relative_to(target_path) 

87 files_to_remove.append(f"{relative_path}/") 

88 except PermissionError as e: 

89 raise DirectoryAccessError( 

90 operation="file_scanning", 

91 directory_path=str(item_path), 

92 message=f"Permission denied: {e}", 

93 details=f"Cannot access file/directory: {item_path}", 

94 ) from e 

95 except Exception as e: 

96 logger.warning(f"Failed to process item {item_path}: {e}") 

97 continue 

98 

99 # Sort to ensure files are removed before their parent directories 

100 # Files first, then directories (reverse alphabetical for nested dirs) 

101 files_to_remove.sort(key=lambda x: (x.endswith("/"), x)) 

102 

103 return files_to_remove 

104 

105 except (DirectoryAccessError, PermissionError) as e: 

106 if isinstance(e, DirectoryAccessError): 

107 raise 

108 else: 

109 raise DirectoryAccessError( 

110 operation="list_generation", 

111 directory_path=str(target_path), 

112 message=f"Permission denied: {e}", 

113 details=f"Cannot access target directory: {target_path}", 

114 ) from e 

115 

116 except (FileScanError, DirectoryAccessError): 

117 # Re-raise our custom exceptions 

118 raise 

119 except Exception as e: 

120 # Wrap unexpected external exceptions 

121 raise FileScanError( 

122 operation="list_generation", 

123 target_path=str(target_path), 

124 message=f"Unexpected error during file list generation: {e}", 

125 details=f"Exception type: {type(e).__name__}", 

126 ) from e

134def verify_files_removed(target_path: Path) -> dict[str, str | bool | int]: 

135 """ 

136 Verify that WOMM files were removed successfully. 

137 

138 Args: 

139 target_path: Target installation directory 

140 

141 Returns: 

142 Dict: Dictionary with success status and details 

143 

144 Raises: 

145 UninstallationVerificationError: If files were not removed successfully 

146 DirectoryAccessError: If directory access fails during verification 

147 """ 

148 try: 

149 # Input validation 

150 if not target_path: 

151 raise VerificationServiceError( 

152 verification_type="removal_verification", 

153 target_path="", 

154 message="Target path cannot be empty", 

155 details="Invalid target path provided for verification", 

156 ) 

157 

158 # Check if we can access the directory for verification 

159 if target_path.exists(): 

160 try: 

161 # Try to access the directory to see if it's accessible 

162 target_path.stat() 

163 except PermissionError as e: 

164 raise DirectoryAccessError( 

165 operation="verification", 

166 directory_path=str(target_path), 

167 message=f"Permission denied during verification: {e}", 

168 details=f"Cannot access directory for verification: {target_path}", 

169 ) from e 

170 except Exception as e: 

171 logger.warning(f"Failed to stat target path {target_path}: {e}") 

172 # If we can't stat it, assume it's not accessible and consider it removed 

173 return { 

174 "success": True, 

175 "message": "Target path not accessible (considered removed)", 

176 } 

177 

178 # Directory exists and is accessible 

179 raise VerificationServiceError( 

180 verification_type="removal_verification", 

181 target_path=str(target_path), 

182 message="WOMM directory still exists after removal", 

183 details=f"Directory path: {target_path}", 

184 ) 

185 else: 

186 return {"success": True, "message": "All WOMM files removed successfully"} 

187 

188 except (VerificationServiceError, DirectoryAccessError): 

189 # Re-raise our custom exceptions 

190 raise 

191 except Exception as e: 

192 # Wrap unexpected external exceptions 

193 raise VerificationServiceError( 

194 verification_type="removal_verification", 

195 target_path=str(target_path), 

196 message=f"File removal verification error: {e}", 

197 details=f"Exception type: {type(e).__name__}", 

198 ) from e

202 """

213 DirectoryAccessError: If directory access fails during verification

216 # Input validation 

217 if not target_path:

219 verification_type="directory_removal", 

220 target_path="", 

221 message="Target path cannot be empty", 

222 details="Invalid target path provided for directory verification", 

223 ) 

224 

225 # Check that target directory is gone 

226 if target_path.exists(): 

227 try:

229 target_path.stat() 

230 except PermissionError as e: 

231 raise DirectoryAccessError( 

232 operation="verification", 

233 directory_path=str(target_path), 

234 message=f"Permission denied during verification: {e}", 

235 details=f"Cannot access directory for verification: {target_path}", 

236 ) from e

239 # If we can't stat it, assume it's not accessible and consider it removed 

240 return {

244 

245 # Directory exists and is accessible 

246 raise VerificationServiceError(

248 target_path=str(target_path),
